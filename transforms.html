@Add(functions)
constexpr inline Matrix rotate_x(
float ang
) {
float c { cosf(ang) };
float s { sinf(ang) };
return {
1, 0, 0, 0,
0, c, -s, 0,
0, s, c, 0,
0, 0, 0, 1
};
}
@End(functions)
