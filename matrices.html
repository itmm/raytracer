inline constexpr bool operator==(
const Matrix &a, const Matrix &b
) {
const float *ai = a._v;
const float *bi = b._v;
const float *ae = ai + 16;
for (; ai < ae; ++ai, ++bi) {
if (! eq(*ai, *bi)) {
return false;
}
}
return true;
}

@Add(functions)
Matrix sub(Matrix a, int r, int c) {
while (r < 3) {
int i = r + 1;
for (int k = 3; k >= 0; --k) {
a(r, k) = a(i, k);
}
r = i;
}
while (c < 3) {
int i = c + 1;
for (int k = 3; k >= 0; --k) {
a(k, c) = a(k, i);
}
c = i;
}
for (int i = 3; i >= 0; --i) {
a(i, 3) = 0;
a(3, i) = 0;
}
return a;
}

@Add(functions)
Matrix inv(
const Matrix &a
) {
Matrix res {};
float d { det4(a) };
if (eq(d, 0)) {
throw std::invalid_argument("inv");
}
for (int r = 3; r >= 0; --r) {
for (int c = 3; c >= 0; --c) {
float cf = cofactor4(
a, r, c
);
res(c, r) = cf / d;
}
}
return res;
}
