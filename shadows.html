@Add(unit-tests) {
Material m;
auto pos { mk_point(0, 0, 0) };
auto eyev { mk_vector(0, 0, -1) };
auto normalv { mk_vector(0, 0, -1) };
Point_Light l { mk_point(0, 0, -10), { 1, 1, 1 } };
bool in_shadow = true;
Color e { 0.1, 0.1, 0.1 };
assert(lighting(m, l, pos, eyev, normalv, in_shadow) == e);
} @End(unit-tests)
@Add(functions)
bool is_shadowed(const World &w, const Tuple &p) {
for (auto &l : w.lights) {
auto v = l->position - p;
float dist = abs(v);
auto dir { norm(v) };
Ray r { p, dir };
auto xs { intersect_world(w, r) };
auto h { hit(xs) };
if (h != xs.end() && h->t < dist) {
return true;
}
}
return false;
}
@End(functions)
@Add(unit-tests) {
auto w = default_world();
assert(! is_shadowed(w, mk_point(0, 10, 0)));
} @End(unit-tests)
@Add(unit-tests) {

} @End(unit-tests)

} @End(unit-tests)
@Add(unit-tests) {

} @End(unit-tests)
